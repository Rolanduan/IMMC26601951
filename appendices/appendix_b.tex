\section{Appendix B: Mathematical Derivations and Algorithm Pseudocode}

\subsection{B.1 Lognormal Distribution Parameter Estimation}

For random variable $\tau \sim \text{LN}(\mu, \sigma^2)$, given sample mean $\bar{t}$ and sample variance $s^2$:

\textbf{Derivation:}
\begin{align*}
\mathbb{E}[\tau] &= e^{\mu + \sigma^2/2} = \bar{t} \\
\text{Var}(\tau) &= (e^{\sigma^2} - 1)e^{2\mu + \sigma^2} = s^2
\end{align*}

Taking logarithms:
\begin{align*}
\mu + \frac{\sigma^2}{2} &= \ln(\bar{t}) \\
2\mu + \sigma^2 &= \ln(s^2 + \bar{t}^2)
\end{align*}

Solving for $\mu$ and $\sigma^2$:
\begin{equation}
\boxed{\hat{\sigma}^2 = \ln(s^2 + \bar{t}^2) - 2\ln(\bar{t})}
\end{equation}
\begin{equation}
\boxed{\hat{\mu} = \ln(\bar{t}) - \frac{1}{2}\hat{\sigma}^2}
\end{equation}

\subsection{B.2 Reliability Calculation for Route}

For route $R = (0, 1, 2, ..., n, 0)$ with independent lognormal travel times:

Total travel time: $T_R = \sum_{i=0}^n \tau_{i,i+1}$ where $\tau_{i,i+1} \sim \text{LN}(\mu_{i,i+1}, \sigma_{i,i+1}^2)$

Reliability:
\begin{equation}
\text{Rel}(R) = \Pr(T_R \leq T_{\max})
\end{equation}

Using Fenton-Wilkinson approximation (sum of lognormals $\approx$ lognormal):
\begin{align}
\mu_R &\approx \frac{\mu_{S^2} - 2\ln(\mu_S)}{2} \\
\sigma_R^2 &\approx \ln(\mu_{S^2}) - 2\ln(\mu_S)
\end{align}
where:
\begin{align*}
\mu_S &= \sum_{i=0}^n e^{\mu_{i,i+1} + \sigma_{i,i+1}^2/2} \\
\mu_{S^2} &= \sum_{i=0}^n e^{2\mu_{i,i+1} + 2\sigma_{i,i+1}^2} + 2\sum_{i<j} e^{\mu_{i,i+1} + \mu_{j,j+1} + (\sigma_{i,i+1}^2 + \sigma_{j,j+1}^2)/2}
\end{align*}

Then:
\begin{equation}
\text{Rel}(R) \approx \Phi\left(\frac{\ln(T_{\max}) - \mu_R}{\sigma_R}\right)
\end{equation}
where $\Phi(\cdot)$ is standard normal CDF.

\subsection{B.3 Benders Cut Derivation}

For two-stage stochastic program:
\begin{align}
\min \quad & c^T x + \theta \\
\text{s.t.} \quad & Ax \geq b \\
& \theta \geq \mathbb{E}_\xi[\mathcal{Q}(x, \xi)]
\end{align}

where $\mathcal{Q}(x, \xi) = \min\{q^T y : Wy \geq h - Tx, y \geq 0\}$.

Let dual of subproblem be: $\max\{\pi^T(h - Tx) : \pi^T W \leq q, \pi \geq 0\}$

If $\pi^{(s)}$ is optimal dual for scenario $s$, Benders optimality cut:
\begin{equation}
\theta \geq \sum_{s \in S} p^{(s)} \pi^{(s)} (h - Tx)
\end{equation}

\textbf{Proof Sketch:} By strong duality, $\mathcal{Q}(x, \xi^{(s)}) = \max_\pi \{\pi^T(h - Tx) : \pi^T W \leq q, \pi \geq 0\}$. For fixed $x$, optimal value equals optimal dual value. Optimality cut enforces $\theta \geq \mathbb{E}[\mathcal{Q}(x, \xi)]$.

\subsection{B.4 ALNS Operator Pseudocode}

\subsubsection{Worst Removal Operator}

\begin{algorithmic}[1]
\PROCEDURE{WorstRemoval}{$x, q$}
\STATE Calculate cost increase $\Delta_i$ for removing customer $i$
\STATE Sort customers by $\Delta_i$ (descending)
\STATE Remove top $q$ customers from solution
\RETURN Modified solution
\ENDPROCEDURE
\end{algorithmic}

\subsubsection{Regret Insertion Operator}

\begin{algorithmic}[1]
\PROCEDURE{RegretInsertion}{$x$, removed\_set}
\WHILE{removed\_set not empty}
\FOR{each customer $i \in$ removed\_set}
\STATE Find best insertion position $pos_1(i)$ with cost $c_1(i)$
\STATE Find second-best insertion position $pos_2(i)$ with cost $c_2(i)$
\STATE Calculate regret: $r_i = c_2(i) - c_1(i)$
\ENDFOR
\STATE Insert customer with maximum regret $r_i$
\STATE Remove inserted customer from removed\_set
\ENDWHILE
\RETURN Modified solution
\ENDPROCEDURE
\end{algorithmic}

\subsubsection{Adaptive Weight Update}

\begin{algorithmic}[1]
\PROCEDURE{UpdateWeights}{$\omega$, success}
\STATE $u_\omega \leftarrow u_\omega + 1$ \COMMENT{Update usage count}
\IF{success is global\_best}
\STATE $s_\omega \leftarrow \rho_1 s_\omega$ \COMMENT{Reward: $\rho_1 = 1.0$}
\ELSEIF{success is improvement}
\STATE $s_\omega \leftarrow \rho_2 s_\omega$ \COMMENT{Reward: $\rho_2 = 0.9$}
\ELSEIF{success is accepted}
\STATE $s_\omega \leftarrow \rho_3 s_\omega$ \COMMENT{Reward: $\rho_3 = 0.8$}
\ELSE
\STATE $s_\omega \leftarrow \rho_4 s_\omega$ \COMMENT{Penalty: $\rho_4 = 0.7$}
\ENDIF
\STATE Recalculate: $p_\omega \leftarrow \frac{s_\omega}{\sum_{\omega'} s_{\omega'}}$
\ENDPROCEDURE
\end{algorithmic}

\subsection{B.5 Clustering Algorithm Pseudocode}

\subsubsection{Reliability-Weighted K-Means}

\begin{algorithmic}[1]
\PROCEDURE{ReliabilityWeightedKMeans}{$V$, $W$, $K$, maxIter}
\STATE Initialize: Assign each customer to nearest warehouse
\FOR{iter = 1 to maxIter}
\FOR{each warehouse $w \in W$}
\STATE Calculate reliability-weighted centroid:
\STATE $\bar{c}_w = \frac{\sum_{i \in \mathcal{C}_w} c_i / \alpha_{iw}}{\sum_{i \in \mathcal{C}_w} 1/\alpha_{iw}}$
\ENDFOR
\FOR{each customer $i \in V$}
\STATE Reassign to warehouse minimizing: $d(c_i, \bar{c}_w) / \alpha_{iw}$
\ENDFOR
\IF{no reassignments}
\STATE break
\ENDIF
\ENDFOR
\RETURN Clusters $\{\mathcal{C}_w\}_{w \in W}$
\ENDPROCEDURE
\end{algorithmic}

\subsection{B.6 Multi-Objective Optimization Algorithm}

\subsubsection{$\epsilon$-Constraint Method}

\begin{algorithmic}[1]
\PROCEDURE{EpsilonConstraint}{$f_1, f_2, f_3$, bounds}
\STATE Calculate individual minima: $f_2^{min}, f_3^{min}$
\STATE Calculate individual maxima: $f_2^{max}, f_3^{max}$
\STATE Discretize: Create grid of $(\epsilon_2, \epsilon_3)$ pairs
\STATE solutions $\leftarrow \emptyset$
\FOR{each $(\epsilon_2, \epsilon_3)$ in grid}
\STATE Solve: $\min f_1$ s.t. $f_2 \leq \epsilon_2, f_3 \leq \epsilon_3$
\IF{feasible}
\STATE Add solution to Pareto set
\ENDIF
\ENDFOR
\STATE Remove dominated solutions
\RETURN Non-dominated Pareto frontier
\ENDPROCEDURE
\end{algorithmic}

\subsection{B.7 Sobol Sensitivity Indices Computation}

For model output $Y = f(X_1, ..., X_k)$:

\textbf{First-Order Effect:}
\begin{equation}
S_i = \frac{\text{Var}_{X_i}(\mathbb{E}_{X_{\sim i}}[Y|X_i])}{\text{Var}(Y)}
\end{equation}

\textbf{Total Effect:}
\begin{equation}
S_{Ti} = 1 - \frac{\text{Var}_{X_{\sim i}}(\mathbb{E}_{X_i}[Y|X_{\sim i}])}{\text{Var}(Y)}
\end{equation}

\textbf{Monte Carlo Estimation:}
\begin{enumerate}
\item Generate two $N \times k$ sample matrices: $A$ and $B$
\item Create matrix $A_B^{(i)}$ where column $i$ from $A$, others from $B$
\item Estimate:
\begin{align*}
\text{Var}(Y) &\approx \frac{1}{N}\sum_{j=1}^N f(A)_j^2 - \left(\frac{1}{N}\sum_{j=1}^N f(A)_j\right)^2 \\
S_i &\approx \frac{\frac{1}{N}\sum_{j=1}^N f(B)_j(f(A_B^{(i)})_j - f(A)_j)}{\text{Var}(Y)}
\end{align*}
\end{enumerate}

We use Saltelli's efficient implementation requiring $N(k+2)$ model evaluations for $k$ parameters.
