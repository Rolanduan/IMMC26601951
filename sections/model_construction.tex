\section{Model Construction}

\subsection{Model I: Deterministic Baseline}

We first formulate the deterministic Capacitated VRP with Time Windows (CVRPTW) using expected travel times $\bar{\tau}_{ij} = \mathbb{E}[\tau_{ij}]$.

\textbf{Objective:}
\begin{equation}
\min \quad z = \sum_{k \in K} \sum_{i,j \in V} \bar{\tau}_{ij} x_{ijk} \label{eq:det_obj}
\end{equation}

\textbf{Constraints:}
\begin{align}
& \sum_{k} \sum_{j \neq i} x_{ijk} = 1, \quad \forall i \in V \setminus \{0\} \label{eq:visit} \
& \sum_{i} d_i y_{ik} \leq C_k, \quad \forall k \label{eq:capacity} \
& t_i + \bar{\tau}_{ij} \leq t_j + M(1-x_{ijk}) \label{eq:time_consistency} \
& e_i \leq t_i \leq l_i, \quad \forall i \label{eq:timewindows}
\end{align}

\subsection{Model II: Stochastic Model with Chance Constraints}

We incorporate uncertainty through chance-constrained programming.

\textbf{Stochastic Objective:}
\begin{equation}
\min \quad \mathbb{E}\left[\sum_{k} \sum_{i,j} \tau_{ij} x_{ijk}\right] \approx \frac{1}{|S|}\sum_{s \in S} \sum_{k,j,i} \tau_{ij}^{(s)} x_{ijk} \label{eq:stochastic_obj}
\end{equation}

\textbf{Reliability Constraint:}
For each route $R_k$:
\begin{equation}
\Pr\left(\sum_{(i,j) \in R_k} \tau_{ij} \leq T_{\max}\right) \geq \alpha \label{eq:reliability}
\end{equation}

Using scenario approximation:
\begin{equation}
\frac{1}{|S|}\sum_{s \in S} \mathbb{I}\left(\sum_{(i,j) \in R_k} \tau_{ij}^{(s)} \leq T_{\max}\right) \geq \alpha \label{eq:scenario_reliability}
\end{equation}

\subsection{Model III: Two-Stage Stochastic Program}

\textbf{First-Stage Decisions:} Route planning $x_{ijk}$ (scenario-independent)

\textbf{Second-Stage Recourse:} Adjusted schedules after uncertainty realization

\textbf{Recourse Function:}
\begin{equation}
\mathcal{Q}(x, \tau^{(s)}) = \min_t \left\{\sum_{k,j,i} \tau_{ij}^{(s)} x_{ijk} + \text{penalty}(t_i^{(s)})\right\}
\end{equation}

\textbf{Master Problem:}
\begin{equation}
\min_x \quad \mathbb{E}_\tau[\mathcal{Q}(x,\tau)]
\end{equation}
